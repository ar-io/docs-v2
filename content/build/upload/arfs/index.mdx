### ArFS SDK

```js
import { ArFS } from "@ardrive/arfs-sdk";

const arfs = new ArFS(turbo);

// Create drive
const drive = await arfs.createDrive({
  name: "My Drive",
  privacy: "public",
});

// Create folder
const folder = await arfs.createFolder({
  name: "Documents",
  parentFolderId: drive.rootFolderId,
  driveId: drive.id,
});

// Upload file
const file = await arfs.uploadFile({
  name: "document.pdf",
  data: fileData,
  parentFolderId: folder.id,
  driveId: drive.id,
});
```
