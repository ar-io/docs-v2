# Turbo SDK

SDK for interacting with Turbo, a fast and efficient data upload service for Arweave. Turbo provides high-performance uploads with instant confirmation and transparent pricing.

## Quick Start

<Tabs items={['Node.js', 'Web (React)', 'Web (HTML)']}>
  <Tab value="Node.js">
    <Steps>
      <Step>
        ### Install the SDK

        ```bash
        npm install @ardrive/turbo-sdk
        ```
      </Step>
      <Step>
        ### Import and configure

        ```javascript
        import { TurboFactory } from '@ardrive/turbo-sdk';

        // Create a Turbo client
        const turbo = TurboFactory.authenticated({
          privateKey: 'your-private-key'
        });
        ```
      </Step>
      <Step>
        ### Upload data

        ```javascript
        // Upload data to Arweave via Turbo
        const uploadResult = await turbo.uploadFile({
          fileStreamFactory: () => fs.createReadStream('path/to/file'),
          fileSizeFactory: () => fs.statSync('path/to/file').size,
          signal: AbortSignal.timeout(60_000),
        });

        console.log('Upload complete:', uploadResult);
        ```
      </Step>
    </Steps>

  </Tab>
  
  <Tab value="Web (React)">
    <Steps>
      <Step>
        ### Install the SDK

        ```bash
        npm install @ardrive/turbo-sdk
        ```
      </Step>
      <Step>
        ### Install polyfills (required for web environments)

        <Callout type="warn">
        Polyfills are required for web environments when using bundlers.
        </Callout>

        ```bash
        npm install --save-dev crypto-browserify stream-browserify buffer
        ```

        Configure your bundler (Webpack, Vite, Rollup, etc.) to provide polyfills for `crypto`, `process`, and `buffer`. Refer to your bundler's documentation for polyfill configuration.
      </Step>
      <Step>
        ### Import and configure

        ```javascript
        import { TurboFactory } from '@ardrive/turbo-sdk/web';

        // Create a Turbo client
        const turbo = TurboFactory.authenticated({
          privateKey: 'your-private-key'
        });
        ```
      </Step>
      <Step>
        ### Create a file upload component

        ```jsx
        import React, { useState } from 'react';
        import { TurboFactory } from '@ardrive/turbo-sdk/web';

        function FileUploader() {
          const [uploading, setUploading] = useState(false);
          const [result, setResult] = useState(null);

          const handleFileUpload = async (event) => {
            const file = event.target.files[0];
            if (!file) return;

            setUploading(true);
            try {
              const turbo = TurboFactory.authenticated({
                privateKey: 'your-private-key'
              });

              const uploadResult = await turbo.uploadFile({
                fileStreamFactory: () => file.stream(),
                fileSizeFactory: () => file.size,
                signal: AbortSignal.timeout(60_000),
              });

              setResult(uploadResult);
              console.log('Upload complete:', uploadResult);
            } catch (error) {
              console.error('Upload failed:', error);
            } finally {
              setUploading(false);
            }
          };

          return (
            <div>
              <input
                type="file"
                onChange={handleFileUpload}
                disabled={uploading}
              />
              {uploading && <p>Uploading...</p>}
              {result && (
                <div>
                  <p>Upload successful!</p>
                  <p>Transaction ID: {result.id}</p>
                </div>
              )}
            </div>
          );
        }

        export default FileUploader;
        ```
      </Step>
    </Steps>

  <Tab value="Web (HTML)">
    <Steps>
      <Step>
        ### Import from CDN

        Use ES module imports directly in your script tags. The bundled SDK includes all necessary polyfills.

        ```html
        <script type="module">
          import { TurboFactory } from 'https://unpkg.com/@ardrive/turbo-sdk@latest/dist/turbo-sdk.esm.js';
          
          // Your code here
        </script>
        ```
      </Step>
      <Step>
        ### Create minimal upload page

        ```html
        <!DOCTYPE html>
        <html lang="en">
        <head>
          <meta charset="UTF-8">
          <meta name="viewport" content="width=device-width, initial-scale=1.0">
          <title>Turbo Upload</title>
        </head>
        <body>
          <h1>Upload to Arweave</h1>
          
          <input type="file" id="fileInput" />
          <button id="uploadBtn" disabled>Upload</button>
          <div id="status"></div>
          
          <script type="module">
            import { TurboFactory } from 'https://unpkg.com/@ardrive/turbo-sdk@latest/dist/turbo-sdk.esm.js';
            
            const fileInput = document.getElementById('fileInput');
            const uploadBtn = document.getElementById('uploadBtn');
            const statusDiv = document.getElementById('status');
            
            fileInput.addEventListener('change', (e) => {
              uploadBtn.disabled = !e.target.files[0];
            });
            
            uploadBtn.addEventListener('click', async () => {
              const file = fileInput.files[0];
              if (!file) return;
              
              statusDiv.textContent = 'Uploading...';
              
              try {
                const turbo = TurboFactory.authenticated({
                  privateKey: 'your-private-key' // Never do this in production!
                });
                
                const result = await turbo.uploadFile({
                  fileStreamFactory: () => file.stream(),
                  fileSizeFactory: () => file.size,
                });
                
                statusDiv.innerHTML = `Success! TX: <a href="https://arweave.net/${result.id}" target="_blank">${result.id}</a>`;
              } catch (error) {
                statusDiv.textContent = `Error: ${error.message}`;
              }
            });
          </script>
        </body>
        </html>
        ```
      </Step>
      <Step>
        ### Use wallet connectors (recommended)

        For production, integrate with wallet providers like Wander to avoid exposing private keys:

        ```html
        <!DOCTYPE html>
        <html lang="en">
        <head>
          <meta charset="UTF-8">
          <meta name="viewport" content="width=device-width, initial-scale=1.0">
          <title>Turbo Upload with Wallet</title>
        </head>
        <body>
          <h1>Upload to Arweave</h1>
          
          <button id="connectBtn">Connect Wallet</button>
          <input type="file" id="fileInput" style="display:none;" />
          <button id="uploadBtn" style="display:none;" disabled>Upload</button>
          <div id="status"></div>
          
          <script type="module">
            import { TurboFactory } from 'https://unpkg.com/@ardrive/turbo-sdk@latest/dist/turbo-sdk.esm.js';
            
            const connectBtn = document.getElementById('connectBtn');
            const fileInput = document.getElementById('fileInput');
            const uploadBtn = document.getElementById('uploadBtn');
            const statusDiv = document.getElementById('status');
            
            // Connect to wallet (ArConnect example)
            connectBtn.addEventListener('click', async () => {
              try {
                // Check if ArConnect is installed
                if (!window.arweaveWallet) {
                  statusDiv.textContent = 'Please install Wander wallet extension';
                  return;
                }
                
                // Connect to wallet
                await window.arweaveWallet.connect(['ACCESS_ADDRESS', 'SIGN_TRANSACTION']);
                const address = await window.arweaveWallet.getActiveAddress();
                
                statusDiv.textContent = `Connected: ${address}`;
                connectBtn.style.display = 'none';
                fileInput.style.display = 'inline';
                uploadBtn.style.display = 'inline';
              } catch (error) {
                statusDiv.textContent = `Connection failed: ${error.message}`;
              }
            });
            
            fileInput.addEventListener('change', (e) => {
              uploadBtn.disabled = !e.target.files[0];
            });
            
            uploadBtn.addEventListener('click', async () => {
              const file = fileInput.files[0];
              if (!file || !turboClient) return;

              if (!window.arweaveWallet) throw new Error('You must connect your wallet!');
              
              statusDiv.textContent = 'Uploading...';

              // Create turbo client using the wallet
              const turboClient =  TurboFactory.authenticated({
                signer: window.arweaveWallet
              });
              
              try {
                const result = await turboClient.uploadFile({
                  fileStreamFactory: () => file.stream(),
                  fileSizeFactory: () => file.size,
                });
                
                statusDiv.innerHTML = `Success! TX: <a href="https://arweave.net/${result.id}" target="_blank">${result.id}</a>`;
              } catch (error) {
                statusDiv.textContent = `Error: ${error.message}`;
              }
            });
          </script>
        </body>
        </html>
        ```
      </Step>
