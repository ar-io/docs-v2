# getArNSRecords

`getArNSRecords` is a method on the `ARIO` class that retrieves all ArNS records with optional pagination and filtering support.

`getArNSRecords` does not require authentication.

## Examples

<CodeGroup title="getArNSRecords">

    ```typescript {{ title: 'NodeJS' }}
    const { ARIO } = require('@ar.io/sdk');

    async function main() {
      const ario = ARIO.init();
      const records = await ario.getArNSRecords({
        limit: 50,
        sortBy: 'endTimestamp',
        sortOrder: 'desc'
      });

      console.log(records);
    }

    main();
    ```

    ```typescript {{ title: 'With Filters' }}
    const { ARIO } = require('@ar.io/sdk');

    async function main() {
      const ario = ARIO.init();
      
      // Filter records by specific names (OR condition)
      const records = await ario.getArNSRecords({
        limit: 50,
        filters: {
          name: ['ardrive', 'ar-io', 'permaweb'] // Returns records with name 'ardrive' OR 'ar-io' OR 'permaweb'
        }
      });

      // Filter by multiple criteria (AND condition)
      const filteredRecords = await ario.getArNSRecords({
        limit: 50,
        filters: {
          name: 'ardrive',           // Must have name 'ardrive'
          type: ['lease', 'perm']    // AND type must be 'lease' OR 'perm'
        }
      });

      console.log(records);
      console.log(filteredRecords);
    }

    main();
    ```
